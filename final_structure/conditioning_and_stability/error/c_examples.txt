/* Square-root test. */
#include <math.h>  // needed for sqrt()
#include <stdio.h>

/* Number of times to apply square root and then square the value. */
#define COUNT 23 

int main() {
  float a = 2.0; 
  int i;

  for (i = 0; i < COUNT; i++)
    a = sqrt(a);   // square root of a

  for (i = 0; i < COUNT; i++)
    a = a * a;     // a squared

  printf("%12g\n", a);

  return 0;
}

/* Can a+b=a for a non-zero b? */ 
#include <stdio.h>

int main() {
  float a = 1.0, b = 0.5, c;

  c = a + b;

  while(c != a) { 
    b = b / 2.0;
    c = a + b;
  }

  printf("%12g %12g %12g\n", a, b, c); 

  return 0;
}
