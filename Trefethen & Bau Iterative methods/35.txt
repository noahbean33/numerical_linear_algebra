We're going to do a lazy GMRES here. First we run Arnoldi, then successively solve the approximate least squares problems using brute force.


include("Krylov.jl")
b = ones(m,1);
(Q,H) = arnoldi(A,b,15);
resnorm = [];
for n = 1:15
    e1 = [1;zeros(n)];
    yn = H[1:n+1,1:n] \ (norm(b)*e1);
    resnorm = [resnorm; norm( H[1:n+1,1:n]*yn - norm(b)*e1 )];
    xn = Q[:,1:n]*yn;
end

semilogy(resnorm,"o-");
xlabel("n"), ylabel("norm of residual");
title("Convergence of GMRES residual");
